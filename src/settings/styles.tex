% Sections
\usepackage{titlesec}
\titleformat{\section}
	{\normalfont\Large\bfseries\color{accent}}
	{}{0em}{}

\titlespacing*{\section}
	{0pt}
	{2\vspaceDefault}
	{\vspaceDefault}

% Lists
% Entries (e.g. Experience or Education)
\usepackage{enumitem}
\setlist[itemize]{
    leftmargin=0pt,
    label={},
    nosep,
    itemsep=2\vspaceDefault,
    before=\relax,
    after=\relax,
}

% Subentries (e.g. Designation responsibilities)
\setlist[itemize, 2]{
	leftmargin=\parindentDefault,
	labelsep=\hspaceDefault,
	label=\textasteriskcentered,
	nosep,
	topsep=\vspaceDefault,
	%rightmargin=.2\linewidth
}
